# Building QGIS on the Atmosphere VM

Install QGIS dependencies

```
apt-get install bison ca-certificates cmake dh-python \
doxygen flex gdal-bin git graphviz grass-dev libexpat1-dev \
libfcgi-dev libgdal-dev libgeos-dev libgsl-dev libpq-dev \
libproj-dev libqca-qt5-2-dev libqca-qt5-2-plugins \
libqt5opengl5-dev libqt5scintilla2-dev libqt5sql5-sqlite \
libqt5svg5-dev libqt5webkit5-dev libqt5xmlpatterns5-dev \
libqwt-qt5-dev libspatialindex-dev libspatialite-dev \
libsqlite3-dev libsqlite3-mod-spatialite lighttpd locales \
ninja-build pkg-config poppler-utils pyqt5-dev \
pyqt5-dev-tools pyqt5.qsci-dev python3-all-dev python3-dev \
python3-future python3-gdal python3-mock python3-nose2 \
python3-psycopg2 python3-pyqt5 python3-pyqt5.qsci \
python3-pyqt5.qtsql python3-pyqt5.qtsvg python3-sip \
python3-sip-dev python3-termcolor python3-yaml \
qt5keychain-dev qtbase5-dev qtpositioning5-dev qtscript5-dev \
qttools5-dev qttools5-dev-tools spawn-fcgi txt2tags xauth \
xfonts-100dpi xfonts-75dpi xfonts-base xfonts-scalable xvfb \
cmake-curses-gui expect
```
